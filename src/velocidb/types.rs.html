10/// Represents errors that can occur in VelociDB.
11#[derive(Debug, Error)]
12pub enum VelociError {
13    /// IO error occurred.
14    #[error("IO error: {0}")]
15    Io(#[from] std::io::Error),
16    
17    /// Custom IO error message.
18    #[error("IO error: {0}")]
19    IoError(String),
20    
21    /// Database file corruption detected.
22    #[error("Database corruption: {0}")]
23    Corruption(String),
24    
25    /// Database is busy (locked).
26    #[error("Database is busy")]
27    Busy,
28    
29    /// Item not found (table, row, etc.).
30    #[error("Not found: {0}")]
31    NotFound(String),
32    
33    /// Constraint violation (e.g., unique key).
34    #[error("Constraint violation: {0}")]
35    ConstraintViolation(String),
36    
37    /// SQL parsing error.
38    #[error("Parse error: {0}")]
39    ParseError(String),
40    
41    /// Transaction error.
42    #[error("Transaction error: {0}")]
43    TransactionError(String),
44    
45    /// Type mismatch during value conversion.
46    #[error("Type mismatch: expected {expected}, got {actual}")]
47    TypeMismatch { expected: String, actual: String },
48    
49    /// Storage engine error.
50    #[error("Storage error: {0}")]
51    StorageError(String),
52    
53    /// Feature not implemented.
54    #[error("Not implemented: {0}")]
55    NotImplemented(String),
56}

61/// Represents a value in a database cell.
62///
63/// `Value` supports standard SQL types: NULL, Integer, Float/Real, Text, and Blob.
64#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
65pub enum Value {
66    /// SQL NULL value.
67    Null,
68    /// 64-bit signed integer.
69    Integer(i64),
70    /// 64-bit floating point number.
71    Float(f64),
72    /// Alias for Float.
73    Real(f64),
74    /// UTF-8 text string.
75    Text(String),
76    /// Binary data.
77    Blob(Vec<u8>),
78}
79
80impl Value {
81    /// Converts the value to an integer if possible.
82    pub fn as_integer(&self) -> Result<i64> {
83        match self {
84            Value::Integer(i) => Ok(*i),
85            _ => Err(VelociError::TypeMismatch {
86                expected: "Integer".to_string(),
87                actual: format!("{:?}", self),
88            }),
89        }
90    }
91
92    /// Converts the value to text if possible.
93    pub fn as_text(&self) -> Result<&str> {
94        match self {
95            Value::Text(s) => Ok(s),
96            _ => Err(VelociError::TypeMismatch {
97                expected: "Text".to_string(),
98                actual: format!("{:?}", self),
99            }),
100        }
101    }
102
103    /// Converts the value to a float if possible.
104    ///
105    /// Handles both `Float`/`Real` variants and converts `Integer` to `f64`.
106    pub fn as_float(&self) -> Result<f64> {
107        match self {
108            Value::Float(f) => Ok(*f),
109            Value::Real(f) => Ok(*f),
110            Value::Integer(i) => Ok(*i as f64),
111            _ => Err(VelociError::TypeMismatch {
112                expected: "Float".to_string(),
113                actual: format!("{:?}", self),
114            }),
115        }
116    }
117    
118    /// Returns the size of the value in bytes when serialized.
119    pub fn size_bytes(&self) -> usize {
120        match self {
121            Value::Null => 1,
122            Value::Integer(_) => 9, // 1 byte type + 8 bytes data
123            Value::Float(_) => 9,   // 1 byte type + 8 bytes data
124            Value::Real(_) => 9,    // 1 byte type + 8 bytes data
125            Value::Text(s) => 1 + 4 + s.len(), // 1 byte type + 4 bytes length + data
126            Value::Blob(b) => 1 + 4 + b.len(), // 1 byte type + 4 bytes length + data
127        }
128    }
129}
130
131impl fmt::Display for Value {
132    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
133        match self {
134            Value::Null => write!(f, "NULL"),
135            Value::Integer(i) => write!(f, "{}", i),
136            Value::Float(fl) => write!(f, "{}", fl),
137            Value::Real(fl) => write!(f, "{}", fl),
138            Value::Text(s) => write!(f, "{}", s),
139            Value::Blob(b) => write!(f, "<blob {} bytes>", b.len()),
140        }
141    }
142}

144/// Supported SQL data types.
145#[derive(Debug, Clone, PartialEq, Eq)]
146pub enum DataType {
147    Integer,
148    Real,
149    Text,
150    Blob,
151    Null,
152}
153
154impl DataType {
155    /// Parses a data type from a string (case-insensitive).
156    pub fn from_str(s: &str) -> Self {
157        match s.to_uppercase().as_str() {
158            "INTEGER" | "INT" => DataType::Integer,
159            "REAL" | "FLOAT" | "DOUBLE" => DataType::Real,
160            "TEXT" | "VARCHAR" | "STRING" => DataType::Text,
161            "BLOB" => DataType::Blob,
162            _ => DataType::Text, // Default to TEXT
163        }
164    }
165}

196/// The result of a SQL query.
197///
198/// Contains the schema (columns) and the data (rows).
199#[derive(Debug)]
200pub struct QueryResult {
201    /// The columns in the result set.
202    pub columns: Vec<Column>,
203    /// The rows returned by the query.
204    pub rows: Vec<Row>,
205}
206
207impl QueryResult {
208    /// Creates a new query result.
209    pub fn new(columns: Vec<Column>, rows: Vec<Row>) -> Self {
210        Self { columns, rows }
211    }
212
213    /// Creates an empty query result.
214    pub fn empty() -> Self {
215        Self {
216            columns: vec![],
217            rows: vec![],
218        }
219    }
