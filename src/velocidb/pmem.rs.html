18/// Direct Access (DAX) mode for persistent memory
19/// Bypasses kernel page cache for direct memory access
20pub struct DaxVfs {
21    /// Memory-mapped file
22    mmap: Arc<RwLock<Option<MmapMut>>>,
23    /// File path
24    path: PathBuf,
25    /// Number of pages
26    num_pages: Arc<AtomicU64>,
27    /// Base address for offset calculations
28    base_addr: Arc<AtomicU64>,
29}
30
31impl DaxVfs {
32    /// Create a new DAX VFS
33    /// The file should be on a DAX-enabled filesystem (e.g., ext4 with dax mount option)
34    pub async fn new(path: impl AsRef<Path>) -> Result<Self> {
35        let path = path.as_ref().to_path_buf();
36
37        // Open file with direct I/O
38        let file = OpenOptions::new()
39            .read(true)
40            .write(true)
41            .create(true)
42            .open(&path)?;
43
44        let metadata = file.metadata()?;
45        let file_size = metadata.len();
46        let actual_size = file_size;
47        let num_pages = (actual_size + PAGE_SIZE as u64 - 1) / PAGE_SIZE as u64;
48
49        // Create memory mapping if file is not empty
50        let mmap = if actual_size > 0 {
51            let mut mmap = unsafe {
52                MmapOptions::new()
53                    .len(actual_size as usize)
54                    .map_mut(&file)?
55            };
56
57            // Advise kernel about usage pattern
58            #[cfg(target_os = "linux")]
59            {
60                use libc::{madvise, MADV_SEQUENTIAL, MADV_WILLNEED};
61                unsafe {
62                    madvise(
63                        mmap.as_mut_ptr() as *mut libc::c_void,
64                        mmap.len(),
65                        MADV_SEQUENTIAL | MADV_WILLNEED,
66                    );
67                }
68            }
69            Some(mmap)
70        } else {
71            None
72        };
73
74        let base_addr = if let Some(ref m) = mmap {
75            m.as_ptr() as u64
76        } else {
77            0
78        };
79
80        Ok(Self {
81            mmap: Arc::new(RwLock::new(mmap)),
82            path,
83            num_pages: Arc::new(AtomicU64::new(num_pages)),
84            base_addr: Arc::new(AtomicU64::new(base_addr)),
85        })
86    }
87
88    /// Get a pointer to a page (zero-copy access)
89    pub fn get_page_ptr(&self, page_id: PageId) -> Result<*const u8> {
90        let num_pages = self.num_pages.load(Ordering::Acquire);
91        
92        if page_id >= num_pages {
93            return Err(VelociError::NotFound(format!("Page {} out of bounds", page_id)));
94        }
95
96        let mmap_guard = self.mmap.read();
97        let mmap = mmap_guard.as_ref().ok_or_else(|| VelociError::StorageError("Memory map not initialized".to_string()))?;
98        let offset = page_id as usize * PAGE_SIZE;
99        
100        Ok(unsafe { mmap.as_ptr().add(offset) })
101    }
102
103    /// Get a mutable pointer to a page (zero-copy writes)
104    pub fn get_page_ptr_mut(&self, page_id: PageId) -> Result<*mut u8> {
105        let num_pages = self.num_pages.load(Ordering::Acquire);
106        
107        if page_id >= num_pages {
108            return Err(VelociError::NotFound(format!("Page {} out of bounds", page_id)));
109        }
110
111        let mut mmap_guard = self.mmap.write();
112        let mmap = mmap_guard.as_mut().ok_or_else(|| VelociError::StorageError("Memory map not initialized".to_string()))?;
113        let offset = page_id as usize * PAGE_SIZE;
114        
115        Ok(unsafe { mmap.as_mut_ptr().add(offset) })
116    }
117
118    /// Persist data using cache line flushes (clflush/clflushopt/clwb)
119    pub fn persist_page(&self, page_id: PageId) -> Result<()> {
120        let ptr = self.get_page_ptr(page_id)?;
121
122        #[cfg(all(target_arch = "x86_64", not(doc)))]
123        {
124            // Use CLWB (Cache Line Write Back) if available, otherwise CLFLUSHOPT
125            if is_x86_feature_detected!("clwb") {
126                unsafe { Self::persist_with_clwb(ptr, PAGE_SIZE) };
127            } else if is_x86_feature_detected!("clflushopt") {
128                unsafe { Self::persist_with_clflushopt(ptr, PAGE_SIZE) };
129            } else {
130                unsafe { Self::persist_with_clflush(ptr, PAGE_SIZE) };
131            }
132        }
133
134        #[cfg(any(not(target_arch = "x86_64"), doc))]
135        {
136            // Fallback for non-x86 or documentation builds
137            // In a real implementation, we might use msync() here
138            std::sync::atomic::fence(Ordering::SeqCst);
139        }
140
141        // Memory fence to ensure persistence
142        std::sync::atomic::fence(Ordering::SeqCst);
143
144        Ok(())
145    }
146
147    /// Persist using CLWB (most efficient - doesn't invalidate cache line)
148    #[cfg(all(target_arch = "x86_64", not(doc)))]
149    #[target_feature(enable = "clwb")]
150    unsafe fn persist_with_clwb(ptr: *const u8, size: usize) {
151        use std::arch::x86_64::*;
152        
153        const CACHE_LINE_SIZE: usize = 64;
154        for i in (0..size).step_by(CACHE_LINE_SIZE) {
155            _mm_clwb(ptr.add(i) as *const u8);
156        }
157    }
158
159    /// Persist using CLFLUSHOPT (optimized flush)
160    #[cfg(all(target_arch = "x86_64", not(doc)))]
161    #[target_feature(enable = "clflushopt")]
162    unsafe fn persist_with_clflushopt(ptr: *const u8, size: usize) {
163        use std::arch::x86_64::*;
164        
165        const CACHE_LINE_SIZE: usize = 64;
166        for i in (0..size).step_by(CACHE_LINE_SIZE) {
167            _mm_clflushopt(ptr.add(i) as *const u8);
168        }
169    }
170
171    /// Persist using CLFLUSH (standard flush)
172    #[cfg(all(target_arch = "x86_64", not(doc)))]
173    unsafe fn persist_with_clflush(ptr: *const u8, size: usize) {
174        use std::arch::x86_64::*;
175        
176        const CACHE_LINE_SIZE: usize = 64;
177        for i in (0..size).step_by(CACHE_LINE_SIZE) {
178            _mm_clflush(ptr.add(i) as *const u8);
179        }
180    }

182    /// Non-temporal store (bypass cache)
183    pub fn non_temporal_store(&self, page_id: PageId, data: &[u8]) -> Result<()> {
184        let ptr = self.get_page_ptr_mut(page_id)?;
185
186        #[cfg(all(target_arch = "x86_64", not(doc)))]
187        unsafe {
188            Self::non_temporal_memcpy(ptr, data.as_ptr(), data.len().min(PAGE_SIZE));
189        }
190
191        #[cfg(any(not(target_arch = "x86_64"), doc))]
192        unsafe {
193            std::ptr::copy_nonoverlapping(data.as_ptr(), ptr, data.len().min(PAGE_SIZE));
194        }
195
196        self.persist_page(page_id)?;
197
198        Ok(())
199    }
200
201    /// Non-temporal memcpy (uses streaming stores)
202    #[cfg(all(target_arch = "x86_64", not(doc)))]
203    #[target_feature(enable = "sse2")]
204    unsafe fn non_temporal_memcpy(dst: *mut u8, src: *const u8, len: usize) {
205        use std::arch::x86_64::*;
206
207        let chunks = len / 16;
208        let remainder = len % 16;
209
210        for i in 0..chunks {
211            let offset = i * 16;
212            let src_ptr = src.add(offset) as *const __m128i;
213            let dst_ptr = dst.add(offset) as *mut __m128i;
214            
215            let data = _mm_loadu_si128(src_ptr);
216            _mm_stream_si128(dst_ptr, data);
217        }
218
219        // Handle remainder with regular copy
220        if remainder > 0 {
221            let offset = chunks * 16;
222            std::ptr::copy_nonoverlapping(src.add(offset), dst.add(offset), remainder);
223        }
224
225        _mm_sfence(); // Ensure stores are visible
226    }

228    /// Expand the file to accommodate more pages
229    fn expand(&self, new_size: u64) -> Result<()> {
230        let file = OpenOptions::new()
231            .read(true)
232            .write(true)
233            .open(&self.path)?;
234
235        file.set_len(new_size)?;
236
237        // Remap
238        let new_mmap = unsafe {
239            MmapOptions::new()
240                .len(new_size as usize)
241                .map_mut(&file)?
242        };
243
244        *self.mmap.write() = Some(new_mmap);
245        self.num_pages.store(new_size / PAGE_SIZE as u64, Ordering::Release);
246
247        Ok(())
248    }
249}

251#[async_trait]
252impl AsyncVfs for DaxVfs {
253    async fn read_page(&self, page_id: PageId) -> Result<Page> {
254        let ptr = self.get_page_ptr(page_id)?;
255        
256        let mut page = Page::new();
257        unsafe {
258            std::ptr::copy_nonoverlapping(ptr, page.data_mut().as_mut_ptr(), PAGE_SIZE);
259        }
260
261        Ok(page)
262    }
263
264    async fn write_page(&self, page_id: PageId, page: &Page) -> Result<()> {
265        let ptr = self.get_page_ptr_mut(page_id)?;
266        
267        unsafe {
268            std::ptr::copy_nonoverlapping(page.data().as_ptr(), ptr, PAGE_SIZE);
269        }
270
271        // Persist to PMEM
272        self.persist_page(page_id)?;
273
274        Ok(())
275    }
276
277    async fn allocate_page(&self) -> Result<PageId> {
278        let current_pages = self.num_pages.load(Ordering::Acquire);
279        let new_page_id = current_pages;
280
281        // Expand file if needed
282        let new_size = (current_pages + 1) * PAGE_SIZE as u64;
283        self.expand(new_size)?;
284
285        // Initialize the page
286        let page = Page::new();
287        self.write_page(new_page_id, &page).await?;
288
289        Ok(new_page_id)
290    }
291
292    async fn flush(&self) -> Result<()> {
293        // For DAX, data is already persistent after cache line flushes
294        // Just ensure all writes are complete
295        std::sync::atomic::fence(Ordering::SeqCst);
296        Ok(())
297    }
298
299    async fn num_pages(&self) -> u64 {
300        self.num_pages.load(Ordering::Acquire)
301    }
302
303    async fn sync(&self) -> Result<()> {
304        let mmap_guard = self.mmap.read();
305        if let Some(mmap) = mmap_guard.as_ref() {
306            mmap.flush()?;
307        }
308        Ok(())
309    }
310}

312/// PMEM-optimized transaction log
313/// Uses direct memory access for ultra-low latency commits
314pub struct PmemTransactionLog {
315    dax_vfs: Arc<DaxVfs>,
316    log_page: PageId,
317    write_offset: Arc<AtomicU64>,
318}
319
320impl PmemTransactionLog {
321    pub async fn new(dax_vfs: Arc<DaxVfs>) -> Result<Self> {
322        let log_page = dax_vfs.allocate_page().await?;
323
324        Ok(Self {
325            dax_vfs,
326            log_page,
327            write_offset: Arc::new(AtomicU64::new(0)),
328        })
329    }
330
331    /// Append a log entry (returns immediately after cache line flush)
332    pub fn append(&self, data: &[u8]) -> Result<u64> {
333        let offset = self.write_offset.fetch_add(data.len() as u64, Ordering::AcqRel);
334        
335        if offset + data.len() as u64 > PAGE_SIZE as u64 {
336            return Err(VelociError::StorageError("Log page full".to_string()));
337        }
338
339        let page_ptr = self.dax_vfs.get_page_ptr_mut(self.log_page)?;
340        
341        unsafe {
342            let write_ptr = page_ptr.add(offset as usize);
343            std::ptr::copy_nonoverlapping(data.as_ptr(), write_ptr, data.len());
344        }
345
346        // Persist just the written cache lines
347        self.dax_vfs.persist_page(self.log_page)?;
348
349        Ok(offset)
350    }
351
352    /// Read log entries
353    pub fn read(&self, offset: u64, length: usize) -> Result<Vec<u8>> {
354        let page_ptr = self.dax_vfs.get_page_ptr(self.log_page)?;
355        
356        if offset + length as u64 > PAGE_SIZE as u64 {
357            return Err(VelociError::StorageError("Read out of bounds".to_string()));
358        }
359
360        let mut buffer = vec![0u8; length];
361        
362        unsafe {
363            let read_ptr = page_ptr.add(offset as usize);
364            std::ptr::copy_nonoverlapping(read_ptr, buffer.as_mut_ptr(), length);
365        }
366
367        Ok(buffer)
368    }
369}

385    #[tokio::test]
386    async fn test_dax_vfs_basic() {
387        let temp_file = NamedTempFile::new().unwrap();
388        let vfs = DaxVfs::new(temp_file.path()).await.unwrap();
389
390        // Allocate a page
391        let page_id = vfs.allocate_page().await.unwrap();
392        assert_eq!(page_id, 0);
393
394        // Write a page
395        let mut page = Page::new();
396        page.data_mut()[0..4].copy_from_slice(&[1, 2, 3, 4]);
397        vfs.write_page(page_id, &page).await.unwrap();
398
399        // Read it back
400        let read_page = vfs.read_page(page_id).await.unwrap();
401        assert_eq!(read_page.data()[0..4], [1, 2, 3, 4]);
402    }
