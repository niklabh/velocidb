15/// Asynchronous Virtual File System trait
16/// Abstracts different I/O backends (standard async, io_uring, cloud storage)
17#[async_trait]
18pub trait AsyncVfs: Send + Sync {
19    /// Read a page asynchronously
20    async fn read_page(&self, page_id: PageId) -> Result<Page>;
21
22    /// Write a page asynchronously
23    async fn write_page(&self, page_id: PageId, page: &Page) -> Result<()>;
24
25    /// Allocate a new page
26    async fn allocate_page(&self) -> Result<PageId>;
27
28    /// Flush all pending writes
29    async fn flush(&self) -> Result<()>;
30
31    /// Get the number of pages
32    async fn num_pages(&self) -> u64;
33
34    /// Sync data to disk
35    async fn sync(&self) -> Result<()>;
36}

38/// Standard Tokio-based async file system
39pub struct TokioVfs {
40    file_path: PathBuf,
41    file: Arc<TokioRwLock<Option<File>>>,
42    num_pages: Arc<RwLock<u64>>,
43}
44
45impl TokioVfs {
46    pub async fn new(path: impl AsRef<Path>) -> Result<Self> {
47        let path = path.as_ref().to_path_buf();
48        
49        // Open or create the file
50        let file = tokio::fs::OpenOptions::new()
51            .read(true)
52            .write(true)
53            .create(true)
54            .open(&path)
55            .await?;
56
57        let metadata = file.metadata().await?;
58        let file_size = metadata.len();
59        let num_pages = if file_size == 0 {
60            0
61        } else {
62            (file_size + PAGE_SIZE as u64 - 1) / PAGE_SIZE as u64
63        };
64
65        Ok(Self {
66            file_path: path,
67            file: Arc::new(TokioRwLock::new(Some(file))),
68            num_pages: Arc::new(RwLock::new(num_pages)),
69        })
70    }
71
72    /// Helper to get file handle
73    async fn get_file(&self) -> Result<File> {
74        let has_file = self.file.read().await.is_some();
75        
76        if !has_file {
77            // Reopen file if needed
78            let file = tokio::fs::OpenOptions::new()
79                .read(true)
80                .write(true)
81                .open(&self.file_path)
82                .await?;
83            
84            let file_clone = file.try_clone().await?;
85            *self.file.write().await = Some(file);
86            Ok(file_clone)
87        } else {
88            // Clone the file descriptor
89            let guard = self.file.read().await;
90            guard.as_ref().unwrap().try_clone().await.map_err(|e| e.into())
91        }
92    }
93}

95#[async_trait]
96impl AsyncVfs for TokioVfs {
97    async fn read_page(&self, page_id: PageId) -> Result<Page> {
98        let num_pages = *self.num_pages.read();
99        
100        if page_id >= num_pages {
101            return Err(VelociError::NotFound(format!(
102                "Page {} out of bounds",
103                page_id
104            )));
105        }
106
107        let mut file = self.get_file().await?;
108        let mut page = Page::new();
109        let offset = page_id * PAGE_SIZE as u64;
110
111        file.seek(std::io::SeekFrom::Start(offset)).await?;
112        file.read_exact(page.data_mut()).await?;
113
114        Ok(page)
115    }
116
117    async fn write_page(&self, page_id: PageId, page: &Page) -> Result<()> {
118        let mut file = self.get_file().await?;
119        let offset = page_id * PAGE_SIZE as u64;
120
121        file.seek(std::io::SeekFrom::Start(offset)).await?;
122        file.write_all(page.data()).await?;
123        file.flush().await?;
124
125        // Update page count
126        let mut num_pages = self.num_pages.write();
127        if page_id >= *num_pages {
128            *num_pages = page_id + 1;
129        }
130
131        Ok(())
132    }
133
134    async fn allocate_page(&self) -> Result<PageId> {
135        let page_id = {
136            let mut num_pages = self.num_pages.write();
137            let page_id = *num_pages;
138            *num_pages += 1;
139            page_id
140        }; // Lock is dropped here before async operation
141
142        // Initialize the page
143        let page = Page::new();
144        self.write_page(page_id, &page).await?;
145
146        Ok(page_id)
147    }
148
149    async fn flush(&self) -> Result<()> {
150        let mut file = self.get_file().await?;
151        file.flush().await?;
152        Ok(())
153    }
154
155    async fn num_pages(&self) -> u64 {
156        *self.num_pages.read()
157    }
158
159    async fn sync(&self) -> Result<()> {
160        let file = self.get_file().await?;
161        file.sync_all().await?;
162        Ok(())
163    }
164}

166/// Async page cache with LRU eviction
167pub struct AsyncPageCache {
168    cache: Arc<RwLock<lru::LruCache<PageId, Arc<Page>>>>,
169    vfs: Arc<dyn AsyncVfs>,
170}
171
172impl AsyncPageCache {
173    pub fn new(capacity: usize, vfs: Arc<dyn AsyncVfs>) -> Self {
174        Self {
175            cache: Arc::new(RwLock::new(
176                lru::LruCache::new(std::num::NonZeroUsize::new(capacity).unwrap()),
177            )),
178            vfs,
179        }
180    }
181
182    /// Read a page with caching
183    pub async fn read_page(&self, page_id: PageId) -> Result<Arc<Page>> {
184        // Check cache first
185        {
186            let mut cache = self.cache.write();
187            if let Some(page) = cache.get(&page_id) {
188                return Ok(Arc::clone(page));
189            }
190        }
191
192        // Cache miss - read from VFS
193        let page = self.vfs.read_page(page_id).await?;
194        let page_arc = Arc::new(page);
195
196        // Add to cache
197        let mut cache = self.cache.write();
198        cache.put(page_id, Arc::clone(&page_arc));
199
200        Ok(page_arc)
201    }
202
203    /// Write a page through cache
204    pub async fn write_page(&self, page_id: PageId, page: Page) -> Result<()> {
205        // Write to VFS first
206        self.vfs.write_page(page_id, &page).await?;
207
208        // Update cache
209        let page_arc = Arc::new(page);
210        let mut cache = self.cache.write();
211        cache.put(page_id, page_arc);
212
213        Ok(())
214    }
215
216    /// Allocate a new page
217    pub async fn allocate_page(&self) -> Result<PageId> {
218        self.vfs.allocate_page().await
219    }
220
221    /// Flush all writes
222    pub async fn flush(&self) -> Result<()> {
223        self.vfs.flush().await
224    }
225
226    /// Get cache statistics
227    pub fn cache_stats(&self) -> CacheStats {
228        let cache = self.cache.read();
229        CacheStats {
230            size: cache.len(),
231            capacity: cache.cap().get(),
232        }
233    }
234}

242/// Async Pager that coordinates I/O operations
243pub struct AsyncPager {
244    cache: Arc<AsyncPageCache>,
245    vfs: Arc<dyn AsyncVfs>,
246}
247
248impl AsyncPager {
249    pub async fn new(path: impl AsRef<Path>, cache_size: usize) -> Result<Self> {
250        let vfs: Arc<dyn AsyncVfs> = Arc::new(TokioVfs::new(path).await?);
251        let cache = Arc::new(AsyncPageCache::new(cache_size, Arc::clone(&vfs)));
252
253        Ok(Self { cache, vfs })
254    }
255
256    /// Read a page
257    pub async fn read_page(&self, page_id: PageId) -> Result<Arc<Page>> {
258        self.cache.read_page(page_id).await
259    }
260
261    /// Write a page
262    pub async fn write_page(&self, page_id: PageId, page: Page) -> Result<()> {
263        self.cache.write_page(page_id, page).await
264    }
265
266    /// Allocate a new page
267    pub async fn allocate_page(&self) -> Result<PageId> {
268        self.cache.allocate_page().await
269    }
270
271    /// Flush all pending writes
272    pub async fn flush(&self) -> Result<()> {
273        self.cache.flush().await?;
274        self.vfs.sync().await
275    }
276
277    /// Get the number of pages
278    pub async fn num_pages(&self) -> u64 {
279        self.vfs.num_pages().await
280    }
281
282    /// Get cache statistics
283    pub fn cache_stats(&self) -> CacheStats {
284        self.cache.cache_stats()
285    }
286}

288/// I/O request for batching
289#[derive(Debug)]
290pub enum IoRequest {
291    Read { page_id: PageId },
292    Write { page_id: PageId, page: Page },
293}
294
295/// Batch I/O executor for improved performance
296pub struct BatchIoExecutor {
297    pager: Arc<AsyncPager>,
298}
299
300impl BatchIoExecutor {
301    pub fn new(pager: Arc<AsyncPager>) -> Self {
302        Self { pager }
303    }
304
305    /// Execute a batch of I/O requests in parallel
306    pub async fn execute_batch(&self, requests: Vec<IoRequest>) -> Result<HashMap<PageId, Page>> {
307        let mut handles = Vec::new();
308        let mut results = HashMap::new();
309
310        for request in requests {
311            let pager = Arc::clone(&self.pager);
312            
313            let handle = tokio::spawn(async move {
314                match request {
315                    IoRequest::Read { page_id } => {
316                        let page = pager.read_page(page_id).await?;
317                        Ok::<_, VelociError>((page_id, (*page).clone()))
318                    }
319                    IoRequest::Write { page_id, page } => {
320                        pager.write_page(page_id, page.clone()).await?;
321                        Ok((page_id, page))
322                    }
323                }
324            });
325
326            handles.push(handle);
327        }
328
329        // Wait for all requests to complete
330        for handle in handles {
331            match handle.await {
332                Ok(Ok((page_id, page))) => {
333                    results.insert(page_id, page);
334                }
335                Ok(Err(e)) => return Err(e),
336                Err(e) => return Err(VelociError::IoError(e.to_string())),
337            }
338        }
339
340        Ok(results)
341    }
342
343    /// Read multiple pages in parallel
344    pub async fn read_batch(&self, page_ids: Vec<PageId>) -> Result<HashMap<PageId, Page>> {
345        let requests: Vec<IoRequest> = page_ids
346            .into_iter()
347            .map(|page_id| IoRequest::Read { page_id })
348            .collect();
349
350        self.execute_batch(requests).await
351    }
352}
